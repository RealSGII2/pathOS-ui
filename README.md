```bash
npm install --save path-os-ui
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'path-os-ui'
import 'path-os-ui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
